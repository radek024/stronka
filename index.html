WYŚWIETLANIE DANYCH Z ŁĄCZENIEM TABEL
select imie, nazwisko, ocena, nazwa from uczniowie as u
join oceny as o on u.id_ucznia=o.id_ucznia join przedmioty as p
on p.id_przedmiotu=o.id_przedmiotu where u.id_ucznia=3;

WYŚWIETLANIE DANYCH Z GRUPOWANIEM I SORTOWANIEM
select imie, nazwisko, nazwa from uczniowie as u join oceny as o
on u.id_ucznia=o.id_ucznia join przedmioty as p
on p.id_przedmiotu=o.id_przedmiotu where u.id_ucznia=3
group by nazwa, imie, nazwisko
order by nazwa;

WYŚWIETLANIE ŚREDNICH OCEN
select imie, nazwisko, avg(ocena) from uczniowie as u
join oceny as o on u.id_ucznia=o.id_ucznia
where u.id_ucznia=3
group by imie, nazwisko;

WYŚWIETLANIE ŚREDNICH OCEN Z ZAOKRĄGLENIEM
select imie, nazwisko, round(avg(ocena),2) from uczniowie
as u join oceny as o on u.id_ucznia=o.id_ucznia
where u.id_ucznia=3
group by imie, nazwisko;

WYSWIETLANIE ŚREDNICH OCEN MALEJĄCO
select imie, nazwisko, round(avg(ocena),2) from nauczyciele
as n join oceny as o on o.id_nauczyciela=n.id_nauczyciela
group by imie, nazwisko
order by round(avg(ocena),2) desc ;
